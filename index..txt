            function setupPagination() {
                document.querySelectorAll('.section-content').forEach(sectionContent => {
                    const sections = {

                        'spo-section': 15,
                        'dpo-section': 15,
                        'other-section': 15

                    const sectionId = sectionContent.id;
                    const rowsPerPage = 5;
                    
                    // Для разделов, кроме первого (ВУЗЫ), загружаем данные
                    if (sectionId !== 'vuz-section' && sectionContent.querySelector('.loading')) {
                        setTimeout(() => {
                            sectionContent.querySelector('.loading').remove();
                            sectionContent.innerHTML = generateSectionHTML(sectionId, sections[sectionId]);
                        }, 500);
                    }
                    
                    // Настраиваем пагинацию только для видимых разделов
                    if (sectionContent.style.display !== 'none') {
                        const table = sectionContent.querySelector('table');
                        if (table) {
                            const tbody = table.querySelector('tbody');
                            const rows = Array.from(tbody.querySelectorAll('tr')).filter(row => 
                                !row.querySelector('td[colspan]') // Исключаем заглушки
                            );
                            const pageCount = Math.ceil(rows.length / rowsPerPage);
                            const pagination = sectionContent.querySelector('.pagination');
                            
                            if (pagination && rows.length > rowsPerPage) {
                                // Очищаем текущую пагинацию и создаем новую
                                pagination.innerHTML = '';

                                const prevBtn = document.createElement('a');

                                prevBtn.classList.add('prev');
                                pagination.appendChild(prevBtn);

                                for (let i = 1; i <= pageCount; i++) {
                                    const pageBtn = document.createElement('a');
                                    pageBtn.href = '#';
                                    pageBtn.textContent = i;
                                    if (i === 1) pageBtn.classList.add('active');
                                    pagination.appendChild(pageBtn);

                                // Добавляем кнопки "следующая" страница
                                const nextBtn = document.createElement('a');
                                nextBtn.href = '#';
                                nextBtn.innerHTML = '&raquo;';
                                nextBtn.classList.add('next');
                                pagination.appendChild(nextBtn);
                                
                                // Показываем первую страницу
                                showPage(1, rows, rowsPerPage, tbody);
                                
                                // Обработчики для кнопок пагинации
                                pagination.addEventListener('click', (e) => {
                                    if (e.target.tagName === 'A' && e.target.href.includes('#')) {
                                        e.preventDefault();
                                        
                                        const activeBtn = pagination.querySelector('a.active');
                                        let page = parseInt(activeBtn.textContent);
                                        
                                        if (e.target.classList.contains('prev')) {
                                            page = Math.max(1, page - 1);
                                        } else if (e.target.classList.contains('next')) {
                                            page = Math.min(pageCount, page + 1);
                                        } else {
                                            page = parseInt(e.target.textContent);
                                        }
                                        
                                        // Обновляем активную кнопку
                                        pagination.querySelectorAll('a').forEach(btn => 
                                            btn.classList.remove('active')
                                        );
                                        
                                        if (e.target.classList.contains('prev') || e.target.classList.contains('next')) {
                                            // Находим новую активную кнопку
                                            const newActiveBtn = pagination.querySelectorAll('a')[page];
                                            if (newActiveBtn && !isNaN(parseInt(newActiveBtn.textContent))) {
                                                newActiveBtn.classList.add('active');
                                            }
                                        } else {
                                            e.target.classList.add('active');
                                        }
                                        
                                        showPage(page, rows, rowsPerPage, tbody);
                                    }
                                });
                            }
                        }
                    }
                });
            }

                // Показываем строки для текущей страницы
                for (let i = start; i < end; i++) {
                    rows[i].style.display = '';
                }
            }

            window.exportToCSV = function(tableId, filename) {
                const table = document.getElementById(tableId);
                if (!table) return;
                
                let csv = [];
                let rows = table.querySelectorAll('tr');
                
                for (let i = 0; i < rows.length; i++) {
                    let row = [], cols = rows[i].querySelectorAll('td, th');
                    
                    for (let j = 0; j < cols.length; j++) {
                        // Убираем HTML теги и обрабатываем спецсимволы
                        let text = cols[j].innerText.replace(/"/g, '""');
                        // Если в тексте есть запятая или кавычки, заключаем в кавычки
                        if (text.includes(',') || text.includes('"') || text.includes('\n')) {
                            text = '"' + text + '"';
                        }
                        row.push(text);
                    }
                    
                    csv.push(row.join(','));
                }
                
                // Создаем ссылку для скачивания
                const csvFile = new Blob([csv.join('\n')], {type: 'text/csv'});
                const downloadLink = document.createElement('a');
                downloadLink.download = filename;
                downloadLink.href = window.URL.createObjectURL(csvFile);
                downloadLink.style.display = 'none';
